\begin{algorithm}
\caption{Adaptive Pruning Routine}
\begin{algorithmic}[1]
\Require Pretrained model $M$, sparsity ratio $\rho$
\Ensure Pruned model $M'$
\State Compute sensitivity score $S_i$ for each layer
\State Sort layers by $S_i$ in ascending order
\State Remove bottom $\rho$ fraction of filters
\State Fine-tune $M'$ for 10 epochs
\end{algorithmic}
\label{alg:train}
\end{algorithm}
